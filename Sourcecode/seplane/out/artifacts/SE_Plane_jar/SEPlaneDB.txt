create table benutzer(
id int not null AUTO_INCREMENT,
vorname varchar(45),
nachname varchar(45),
benutzername varchar(45),
passwort char(100),
passwort_klar varchar(45),
benutzertyp ENUM('admin', 'fgm', 'kunde'),
primary key(id)
);

create table fluggesellschaft(
id int not null AUTO_INCREMENT,
fgmanager int,
name varchar(45),
land varchar(45),
budget double,
primary key (id),
foreign key (fgmanager) references benutzer(id));

create table flughafen(
id char(5) not null auto_increment,
ort varchar(45),
land varchar(45),
name varchar(100),
runway int,
satzung double,
latitude double,
logitude double,
primary key (id));

create table flugzeug(
id int not null AUTO_INCREMENT,
hersteller varchar(45),
flugzeugtyp varchar(45),
preis double,
reichweite double,
sitzplaetze int,
primary key (id));

create table fluglinie(
id int not null auto_increment,
start char(5),
ziel char(5),
startdatum date,
intervall enum('tag','woche','monat'),
fluggesellschaft int,
flugzeug int,
flugnummer int,
anze int,
anzb int,
preisee double,
preiseb double,
primary key (id),
foreign key (start) references flughafen(id),
foreign key (ziel) references flughafen(id),
foreign key (fluggesellschaft) references fluggesellschaft(id),
foreign key (flugzeug) references flugzeug(id));

create table zuordnung_flugzeug(
fg_id int not null,
f_id int not null,
primary key(fg_id, f_id),
foreign key (fg_id) references fluggesellschaft(id),
foreign key (f_id) references flugzeug(id));
